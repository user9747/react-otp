# react-otp-box

> Made with create-react-library

## Install

```bash
npm install --save react-otp-box
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-otp-box'
import 'react-otp-box/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
